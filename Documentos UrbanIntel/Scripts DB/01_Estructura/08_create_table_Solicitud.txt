CREATE TABLE Solicitud (
    id INT PRIMARY KEY AUTO_INCREMENT,
    rut_usuario VARCHAR(12),         
    direccion TEXT NOT NULL,
    descripcion TEXT NOT NULL,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,  -- Se genera al crear la solicitud
    fecha_aprobacion DATETIME,                           -- Cuando un funcionario la aprueba
    fecha_asignacion DATETIME,                           -- Cuando se asigna a reparacion
    rut_ciudadano VARCHAR(12), 
	telefono_ciudadano VARCHAR(20),
    email_ciudadano VARCHAR(100),
    tipo_reparacion_id INT,
    prioridad_id INT,
    estado_id INT,
    tiempo_estimado VARCHAR(50),
    FOREIGN KEY (rut_usuario) REFERENCES Usuario(rut),
    FOREIGN KEY (tipo_reparacion_id) REFERENCES TipoReparacion(id),
    FOREIGN KEY (prioridad_id) REFERENCES Prioridad(id),
    FOREIGN KEY (estado_id) REFERENCES EstadoSolicitud(id)
);
